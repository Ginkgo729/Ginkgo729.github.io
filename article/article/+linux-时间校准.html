方式一 

tzselect
export TZ='Asia/Shanghai'
source ~/.bashrc
rm -f /etc/localtime
ln -s /usr/share/zoneinfo/Asia/Shanghai /etc/localtime
timedatectl set-timezone 'Asia/Shanghai'

yum -y install ntp
/usr/sbin/ntpdate -u cn.pool.ntp.org
